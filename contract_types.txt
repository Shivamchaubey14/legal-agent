class CustomUser(AbstractBaseUser, PermissionsMixin):
    ROLE_CHOICES = [
        ('data_entry', 'Data Entry'),
        ('viewer', 'Viewer'),
    ]
    
    ASSIGNED_UNDER_CHOICES = [
        ('legal', 'Legal'),
        ('logistic inward', 'Logistic Inward'),
        ('logistic outward', 'Logistic Outward'),
    ]
    
    email = models.EmailField(unique=True)
    first_name = models.CharField(max_length=250, null=True)
    last_name = models.CharField(max_length=250, null=True)
    role = models.CharField(max_length=20, choices=ROLE_CHOICES)
    assigned_under = models.CharField(
        max_length=20, 
        choices=ASSIGNED_UNDER_CHOICES,
        null=True,
        blank=True,
        help_text="Department assignment for data entry users"
    )
    is_active = models.BooleanField(default=True)
    is_staff = models.BooleanField(default=False)
    employee_code = models.CharField(max_length=250, null=True)
    profile_picture = models.ImageField(
        upload_to='profile_pics/',
        null=True,
        blank=True,
        help_text="Upload a profile picture (JPG, PNG, Max 2MB)"
    )

    USERNAME_FIELD = 'email'
    REQUIRED_FIELDS = ['role']

    objects = CustomUserManager()

    def __str__(self):
        return self.email
        
    def can_access_agreement_type(self, agreement_type):
        """Check if user can access specific agreement type based on role and assignment"""
        if self.role != 'data_entry':
            return True  # Non-data entry users can see all
            
        if self.assigned_under == 'legal':
            legal_agreements = [
                'amc', 'bmc', 'input_services', 'consultant', 'cook', 
                'distributer', 'milk_sale', 'mcc', 'mpacs', 'mpp_sahayak', 
                'rta', 'godown', 'office_lease', 'guest_house'
            ]
            return agreement_type in legal_agreements
            
        elif self.assigned_under == 'logistic inward':
            logistic_agreements = ['inward']
            return agreement_type in logistic_agreements
        elif self.assigned_under == 'logistic outward':
            logistic_agreements = ['outward']
            return agreement_type in logistic_agreements            
        return False
    
    def can_assign_view_rights(self):
        """Check if this user can assign view rights to others"""
        can_assign = self.email == 'nikhil.agarwal@shwetdharamilk.com'
        print(f"=== CHECKING ASSIGNMENT PERMISSION ===")
        print(f"User: {self.email}")
        print(f"Can assign view rights: {can_assign}")
        return can_assign

    def can_view_agreement_type(self, agreement_type):
        """
        Enhanced method to check if user can view specific agreement type
        This replaces the existing can_access_agreement_type method logic
        """
        print(f"=== CHECKING AGREEMENT ACCESS ===")
        print(f"User: {self.email}")
        print(f"Agreement Type: {agreement_type}")
        print(f"User Role: {self.role}")
        print(f"User Assignment: {self.assigned_under}")
        
        # Use the ViewAssignment model to check permissions
        return ViewAssignment.can_user_view_agreement(self, agreement_type)

    def get_assigned_agreement_types(self):
        """Get list of agreement types this user has been assigned to view"""
        if self.email == 'nikhil.agarwal@shwetdharamilk.com':
            # Nikhil can see all agreement types
            return [choice[0] for choice in ViewAssignment.AGREEMENT_TYPE_CHOICES]

            assigned_user=self,

class UserActionLog(models.Model):
    user = models.ForeignKey(CustomUser, on_delete=models.SET_NULL, null=True, blank=True)
    path = models.CharField(max_length=500)
    method = models.CharField(max_length=10)
    timestamp = models.DateTimeField(auto_now_add=True)
    ip_address = models.GenericIPAddressField(null=True, blank=True)
    user_agent = models.TextField(null=True, blank=True)
    data = models.TextField(null=True, blank=True)  # JSON or key-value string of changes
    changes = models.TextField(null=True, blank=True)  # Human-readable or JSON field describing what changed

    def __str__(self):
        return f"{self.user} - {self.method} - {self.path}"

    def get_local_timestamp(self):
        return localtime(self.timestamp, IST).strftime("%d-%m-%Y %I:%M %p")  # IST format

        if self.actual_completion_date and self.status not in ['COMPLETED', 'EXEMPTED', 'CANCELLED']:
            self.status = 'COMPLETED'
        elif not self.actual_completion_date and self.is_overdue and self.status == 'PENDING':

        """Calculate days overdue (0 if not overdue)"""
        today = datetime.now().date()
        if self.due_date >= today:
            return 0
        return (today - self.due_date).days

class ComplianceType(models.Model):
    """Model to store different types of compliances with dynamic configuration"""
    name = models.CharField(max_length=200, unique=True)

        null=True, 
        blank=True,
        help_text="Day of month when compliance is due (1-31). For monthly/quarterly/annual compliances."
    )
    due_days_after_period_end = models.IntegerField(
        default=0,
        help_text="Number of days after period end when compliance is due"
    )
    
    # Email configuration
    send_reminder_emails = models.BooleanField(default=True)
    reminder_days_before = models.IntegerField(

        help_text="Send reminder email X days before due date"
    )
    escalation_days_after = models.IntegerField(
        default=3,
        help_text="Send escalation email X days after due date"
    )
    
    # Custom period configuration (for custom frequency)
    custom_period_months = models.IntegerField(
        null=True,
        blank=True,
        help_text="For custom frequency: number of months in the period"
    )

    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)
    created_by = models.ForeignKey(
        CustomUser,
        on_delete=models.SET_NULL,
        null=True,
        blank=True
    )
    
    class Meta:
        verbose_name = "Compliance Type"
        verbose_name_plural = "Compliance Types"
        ordering = ['name']
    
    def __str__(self):
        if self.frequency and hasattr(self.frequency, 'display_name'):
            return f"{self.name} ({self.frequency.display_name})"
        else:
            return f"{self.name} (No Frequency Set)"
    
    def calculate_due_date(self, period_start_date, period_end_date=None):
        """Calculate due date for this compliance type based on period"""
        from datetime import timedelta
        from calendar import monthrange
        
        if self.frequency.name == 'MONTHLY':
            # For monthly: due on specific day of next month
            if self.due_day_of_month:
                year = period_start_date.year
                month = period_start_date.month + 1
                if month > 12:
                    month = 1
                    year += 1
                
                # Handle cases where due day doesn't exist in the month (e.g., 31st in February)
                max_day = monthrange(year, month)[1]
                due_day = min(self.due_day_of_month, max_day)
                
                return period_start_date.replace(year=year, month=month, day=due_day)
            else:
                # Default: due_days_after_period_end after month end
                if period_end_date:
                    return period_end_date + timedelta(days=self.due_days_after_period_end)
                else:
                    # Calculate month end
                    year = period_start_date.year
                    month = period_start_date.month + 1
                    if month > 12:
                        month = 1
                        year += 1
                    month_end = period_start_date.replace(year=year, month=month, day=1) - timedelta(days=1)
                    return month_end + timedelta(days=self.due_days_after_period_end)
        
        elif self.frequency.name in ['QUARTERLY', 'HALF_YEARLY', 'ANNUALLY']:
            # For these frequencies, period_end_date should be provided
            if period_end_date:
                return period_end_date + timedelta(days=self.due_days_after_period_end)
            else:
                # Fallback calculation
                return period_start_date + timedelta(days=self.due_days_after_period_end)
        
        elif self.frequency.name == 'CUSTOM':
            # For custom frequency
            if period_end_date:
                return period_end_date + timedelta(days=self.due_days_after_period_end)
            else:
                # Use custom_period_months to calculate
                months_to_add = self.custom_period_months or 1
                year = period_start_date.year
                month = period_start_date.month + months_to_add
                while month > 12:
                    month -= 12
                    year += 1
                period_end = period_start_date.replace(year=year, month=month) - timedelta(days=1)
                return period_end + timedelta(days=self.due_days_after_period_end)
        
        # Default fallback
        return period_start_date + timedelta(days=30)

        blank=True,
        related_name='created_compliances'
    )
    completed_by = models.ForeignKey(
        CustomUser,
        on_delete=models.SET_NULL,
        null=True,
        blank=True,
        related_name='completed_compliances'
    )
    initial_notification_sent = models.DateTimeField(
        blank=True, 
        null=True,
        help_text="Timestamp when initial compliance notification was sent"
    )
    
    class Meta:
        verbose_name = "Compliance"
        verbose_name_plural = "Compliances"
        unique_together = ['compliance_type', 'period']
        ordering = ['-due_date', 'compliance_type__name']
    
    def __str__(self):
        return f"{self.compliance_type.name} - {self.period.display_name}"
    
    def save(self, *args, **kwargs):
        # Generate unique tracking ID if not exists
        if not self.email_tracking_id:
            self.email_tracking_id = f"COMP-{uuid.uuid4().hex[:8]}-{self.compliance_type.id}-{self.period.id}"
        
        # Auto-update status based on completion
        if self.actual_completion_date and self.status not in ['COMPLETED', 'EXEMPTED', 'CANCELLED']:
            self.status = 'COMPLETED'
        elif not self.actual_completion_date and self.is_overdue and self.status == 'PENDING':
            self.status = 'OVERDUE'
            
        super().save(*args, **kwargs)
    
    @property
    def is_overdue(self):
        """Check if compliance is overdue"""
        if self.status in ['COMPLETED', 'EXEMPTED', 'CANCELLED']:
            return False
        return self.due_date < datetime.now().date()
    
    @property
    def days_until_due(self):
        """Calculate days until due date"""
        today = datetime.now().date()
        delta = self.due_date - today
        return delta.days
    
    @property
    def days_overdue(self):
        """Calculate days overdue (0 if not overdue)"""
        today = datetime.now().date()
        if self.due_date >= today:
            return 0
        return (today - self.due_date).days
    
    def mark_completed(self, user=None, document=None, completion_date=None, remarks=None):
        """Mark compliance as completed"""
        self.status = 'COMPLETED'
        self.actual_completion_date = completion_date or datetime.now().date()
        self.completed_by = user
        if document:
            self.compliance_document = document
        if remarks:
            self.remarks = remarks
        self.save()
    
    def send_reminder_email(self):
        """Send reminder email for this compliance"""
        try:
            from .management.commands.compliance_email_manager import ComplianceEmailManager
            email_manager = ComplianceEmailManager()
            return email_manager.send_compliance_reminder(self)
        except ImportError:
            return False
    
    def send_escalation_email(self):
        """Send escalation email for overdue compliance"""
        try:
            from .management.commands.compliance_email_manager import ComplianceEmailManager
            email_manager = ComplianceEmailManager()
            return email_manager.send_compliance_escalation(self)
        except ImportError:
            return False

class ComplianceTemplate(models.Model):
    """Model to store compliance templates for bulk creation"""
    
    name = models.CharField(max_length=200, unique=True)
    description = models.TextField(blank=True, null=True)
    
    # Template configuration
    compliance_types = models.ManyToManyField(
        ComplianceType,
        related_name='templates',
        help_text="Compliance types included in this template"
    )
    
    # Auto-generation settings
    auto_generate_for_year = models.BooleanField(
        default=False,
        help_text="Automatically generate all periods for a year"
    )
    
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)
    created_by = models.ForeignKey(
        CustomUser,
        on_delete=models.SET_NULL,
        null=True,
        blank=True
    )
    
    class Meta:
        verbose_name = "Compliance Template"
        verbose_name_plural = "Compliance Templates"
        ordering = ['name']
    
    def __str__(self):
        return self.name
    
    def generate_compliances_for_year(self, year, user=None):
        """Generate all compliances for a given year based on this template"""
        from datetime import date
        from calendar import monthrange
        
        created_compliances = []
        
        for compliance_type in self.compliance_types.all():
            periods_created = []
            
            if compliance_type.frequency.name == 'MONTHLY':
                # Create 12 monthly compliances
                for month in range(1, 13):
                    # Create or get period
                    period, created = CompliancePeriod.objects.get_or_create(
                        period_type='MONTHLY',
                        year=year,
                        month=month,
                        defaults={
                            'start_date': date(year, month, 1),
                            'end_date': date(year, month, monthrange(year, month)[1])
                        }
                    )
                    periods_created.append(period)
                    
            elif compliance_type.frequency.name == 'QUARTERLY':
                # Create 4 quarterly compliances
                quarters = [
                    (1, 1, 3),  # Q1: Jan-Mar
                    (2, 4, 6),  # Q2: Apr-Jun
                    (3, 7, 9),  # Q3: Jul-Sep
                    (4, 10, 12)  # Q4: Oct-Dec
                ]
                
                for quarter, start_month, end_month in quarters:
                    period, created = CompliancePeriod.objects.get_or_create(
                        period_type='QUARTERLY',
                        year=year,
                        quarter=quarter,
                        defaults={
                            'start_date': date(year, start_month, 1),
                            'end_date': date(year, end_month, monthrange(year, end_month)[1])
                        }
                    )
                    periods_created.append(period)
                    
            elif compliance_type.frequency.name == 'HALF_YEARLY':
                # Create 2 half-yearly compliances
                halves = [
                    (1, 1, 6),   # H1: Jan-Jun
                    (2, 7, 12)   # H2: Jul-Dec
                ]
                
                for half, start_month, end_month in halves:
                    period, created = CompliancePeriod.objects.get_or_create(
                        period_type='HALF_YEARLY',
                        year=year,
                        month=start_month,  # Use month to distinguish H1/H2
                        defaults={
                            'start_date': date(year, start_month, 1),
                            'end_date': date(year, end_month, monthrange(year, end_month)[1]),
                            'display_name': f"H{half} {year}"
                        }
                    )
                    periods_created.append(period)
                    
            elif compliance_type.frequency.name == 'ANNUALLY':
                # Create 1 annual compliance
                period, created = CompliancePeriod.objects.get_or_create(
                    period_type='ANNUALLY',
                    year=year,
                    defaults={
                        'start_date': date(year, 1, 1),
                        'end_date': date(year, 12, 31),
                        'display_name': f"FY {year}"
                    }
                )
                periods_created.append(period)
            
            # Create compliance records for each period
            for period in periods_created:
                due_date = compliance_type.calculate_due_date(period.start_date, period.end_date)
                
                compliance, created = Compliance.objects.get_or_create(
                    compliance_type=compliance_type,
                    period=period,
                    defaults={
                        'due_date': due_date,
                        'created_by': user
                    }
                )
                
                if created:
                    created_compliances.append(compliance)
        
        return created_compliances
